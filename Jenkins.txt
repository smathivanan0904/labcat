pipeline {
    agent any
    stages {
        stage('Version 4 Tasks') {
            steps {
                // Task 1: Checkout
                checkout scm 
                
                script {
                    // Task 2: Read current content and append new message
                    // Note: 'readFile' and 'writeFile' work on both Windows and Linux
                    def currentContent = readFile('status.txt')
                    writeFile file: 'status.txt', text: currentContent + "\nPipeline Execution Completed"
                }
                
                // Task 3: Display contents (Windows command is 'type')
                bat 'type status.txt' 
            }
        }
    }
}
