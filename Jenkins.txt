pipeline {
    agent any
    parameters {
        // Task 2: Define two parameters
        string(name: 'APP_NAME', defaultValue: 'PaymentGateway', description: 'Name of the App')
        string(name: 'VERSION', defaultValue: '3.1.0', description: 'Version number')
    }
    stages {
        stage('V3 Tasks') {
            steps {
                // Task 1: Checkout latest version
                checkout scm
                // Task 3: Combined message
                echo "Deploying Application: ${params.APP_NAME} | Release Version: ${params.VERSION}"
            }
        }
    }
}
