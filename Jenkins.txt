pipeline {
    agent any
    stages {
        stage('Version 4 Tasks') {
            steps {
                checkout scm // Task 1
                script {
                    // Task 2: Read current content and append new message
                    def currentContent = readFile('status.txt')
                    writeFile file: 'status.txt', text: currentContent + "\nPipeline Execution Completed"
                }
                // Task 3: Display contents of status.txt
                sh 'cat status.txt' // Use 'bat type status.txt' if on Windows
            }
        }
    }
}
