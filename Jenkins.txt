pipeline {
    agent any
    parameters {
        // Task 2: Define a numeric parameter
        string(name: 'BUILD_ID', defaultValue: '101', description: 'Enter Numeric Build ID')
    }
    stages {
        stage('V2 Tasks') {
            steps {
                // Task 1: Checkout
                checkout scm
                // Task 3: Save value inside buildinfo.txt
                writeFile file: 'buildinfo.txt', text: "BUILD_ID=${params.BUILD_ID}"
                // Verification: Print content to console
                sh 'cat buildinfo.txt' 
            }
        }
    }
}
